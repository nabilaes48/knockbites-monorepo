## Database Schema Additions Required

### Phase 1 (Analytics)
```sql
-- Daily aggregated metrics
CREATE TABLE daily_metrics (
  id SERIAL PRIMARY KEY,
  store_id INT REFERENCES stores(id),
  date DATE NOT NULL,
  total_orders INT,
  total_revenue DECIMAL(10,2),
  avg_order_value DECIMAL(10,2),
  new_customers INT,
  returning_customers INT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Hourly metrics for heatmaps
CREATE TABLE hourly_metrics (
  id SERIAL PRIMARY KEY,
  store_id INT REFERENCES stores(id),
  date DATE NOT NULL,
  hour INT NOT NULL,
  order_count INT,
  revenue DECIMAL(10,2)
);
```

### Phase 2 (Multi-Location)
```sql
-- Organization hierarchy
CREATE TABLE organizations (
  id SERIAL PRIMARY KEY,
  name VARCHAR(255) NOT NULL,
  plan VARCHAR(50) DEFAULT 'starter',
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Regions for grouping stores
CREATE TABLE regions (
  id SERIAL PRIMARY KEY,
  org_id INT REFERENCES organizations(id),
  name VARCHAR(255) NOT NULL,
  manager_id UUID REFERENCES auth.users(id)
);

-- Link stores to organizations
ALTER TABLE stores ADD COLUMN org_id INT REFERENCES organizations(id);
ALTER TABLE stores ADD COLUMN region_id INT REFERENCES regions(id);
```

### Phase 3 (AI)
```sql
-- Predictions storage
CREATE TABLE predictions (
  id SERIAL PRIMARY KEY,
  store_id INT REFERENCES stores(id),
  prediction_type VARCHAR(50), -- 'sales', 'demand', 'churn'
  target_date DATE,
  predicted_value DECIMAL(10,2),
  confidence DECIMAL(3,2),
  actual_value DECIMAL(10,2),
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Customer segments
CREATE TABLE customer_segments (
  id SERIAL PRIMARY KEY,
  name VARCHAR(255),
  criteria JSONB,
  customer_count INT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
```

### Phase 4 (Marketing)
```sql
-- Email campaigns
CREATE TABLE campaigns (
  id SERIAL PRIMARY KEY,
  org_id INT REFERENCES organizations(id),
  name VARCHAR(255),
  type VARCHAR(50), -- 'email', 'sms', 'push'
  status VARCHAR(50), -- 'draft', 'scheduled', 'sent'
  subject VARCHAR(255),
  content TEXT,
  segment_id INT REFERENCES customer_segments(id),
  scheduled_at TIMESTAMPTZ,
  sent_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Campaign metrics
CREATE TABLE campaign_metrics (
  id SERIAL PRIMARY KEY,
  campaign_id INT REFERENCES campaigns(id),
  sent_count INT,
  delivered_count INT,
  opened_count INT,
  clicked_count INT,
  converted_count INT,
  revenue_attributed DECIMAL(10,2)
);
```

### Phase 5 (Loyalty)
```sql
-- Loyalty tiers
CREATE TABLE loyalty_tiers (
  id SERIAL PRIMARY KEY,
  org_id INT REFERENCES organizations(id),
  name VARCHAR(50),
  min_points INT,
  points_multiplier DECIMAL(3,2),
  benefits JSONB
);

-- Rewards catalog
CREATE TABLE rewards (
  id SERIAL PRIMARY KEY,
  org_id INT REFERENCES organizations(id),
  name VARCHAR(255),
  description TEXT,
  points_cost INT,
  reward_type VARCHAR(50), -- 'free_item', 'discount', 'percentage'
  reward_value DECIMAL(10,2),
  active BOOLEAN DEFAULT true
);

-- Referrals
CREATE TABLE referrals (
  id SERIAL PRIMARY KEY,
  referrer_id UUID REFERENCES auth.users(id),
  referee_id UUID REFERENCES auth.users(id),
  referral_code VARCHAR(20),
  status VARCHAR(50), -- 'pending', 'completed'
  referrer_reward_id INT REFERENCES rewards(id),
  referee_reward_id INT REFERENCES rewards(id),
  created_at TIMESTAMPTZ DEFAULT NOW()
);
```
